State Pattern

Code


function On(){
  this.click = function(context){
    console.log("Turn Off");
    context.setState(new Off());
  }
}

function Off(){
  this.click = function(){
    console.log("Turn On");
    context.setState(new On());
  }

}

function Context(){
  this.current = new Off();

  this.setState = function(state){
    this.current = state;
  };

  this.click = function(){
    this.current.click(context);
  }
}

Usage


var context = new Context();
context.click();
context.click();
context.click();
context.click();
